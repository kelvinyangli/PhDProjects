cpts = read.dsc("Known BNs/asia.dsc")
dag = model2network(modelstring(cpts))
saveRDS(dag, "Known BNs/asiaDag.rds")
allNodes = nodes(dag)
v = rep(0, length(allNodes))
for (i in 1:length(allNodes)) v[i] = nrow(cpts[[i]]$prob)
saveRDS(v, "Known BNs/asiaArity.rds")
dagMatrix = dag2matrix(dag)
write.table(dagMatrix, "pcmb/asia.data.net")
